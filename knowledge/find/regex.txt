# find — regex examples

Task: Find all files/directories matching the regex .*sql.*
Command: find -regex .*sql.*

Task: Find all leaf directories that include only one occurrence of "modules"
Command: find -regex '.*/modules\(/.*\|$\)' \! -regex '.*/modules/.*/modules\(/.*\|$\)' -type d -links 2

Task: Find files in the current directory tree whose names match regular expression "^.*~$\|^.*#$"
Command: find -regex "^.*~$\|^.*#$"

Task: Remove files whose names match regular expression '^.*/[A-Za-z]+-[0-9]+x[0-9]+\.[A-Za-z]+$' from the current directory tree
Command: find -regex '^.*/[A-Za-z]+-[0-9]+x[0-9]+\.[A-Za-z]+$' | xargs echo rm -f

Task: display all files in current folder ending with "~" or "#" using regular expression
Command: find -regex "^.*~$\|^.*#$"

Task: display all files in current folder using regular expression
Command: find -regex "$rx"

Task: find all c, cpp files in current folder
Command: find -regex '.*\.\(c\|cpp\)'
