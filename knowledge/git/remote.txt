# git — Remote and tag operations

List remotes: git remote -v
Add remote: git remote add origin <url>
Change remote URL: git remote set-url origin <url>
Remove remote: git remote remove <name>
Rename remote: git remote rename <old> <new>

Create annotated tag: git tag -a v1.0.0 -m "version 1.0.0"
Create lightweight tag: git tag v1.0.0
List tags: git tag -l
Delete local tag: git tag -d v1.0.0
Delete remote tag: git push origin --delete v1.0.0
Push tag: git push origin v1.0.0
Push all tags: git push origin --tags
